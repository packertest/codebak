filename = "test.txt"

def get_directory_paths(search_text):
    with open(filename, "r") as file:
        content = file.read()

    directory_paths = []
    index = 0

    while index < len(content):
        search_index = content.find(search_text, index)
        if search_index != -1:
            closest_match = content.rfind("Directory of", 0, search_index)
            if closest_match != -1:
                newline_index = content.find("\n", closest_match)
                if newline_index != -1:
                    directory_path = content[closest_match:newline_index].strip()
                    directory_paths.append(directory_path.split("Directory of ")[1])
            index = search_index + len(search_text)
        else:
            break

    return directory_paths

desktop_paths = get_directory_paths("hello")
print("Directory paths for 'hello':")
for path in desktop_paths:
    print(path)

known_game_list_paths = get_directory_paths("KnownGameList.bin")
print("Directory paths for 'KnownGameList.bin':")
for path in known_game_list_paths:
    print(path)
